# Makefile for UNIX 

# the compiler: gcc for C program, define as g++ for C++
CC = gcc 

# compiler flags: 
# -g enable debugging 
# -Wall turn on all warnings 
# -D__USE_FIXED_PROTOTYPES__ force the compiler to use the correct headers 
# -ansi donâ€™t use GNU extensions, stick to ANSI 
# -std=c99 use newer version 
CFLAGS=-g -Wall -D__USE_FIXED_PROTOTYPES__ -std=c99 

# libraries 
# OS X 
OS_X_LIBS += -framework ApplicationServices 
# custom
COCOA_HELPER_LIBRARY = /Users/michzio/Developer/MyProjects/BachelorDegree/c/test/OS_X/cocoa_helper/libcocoa_helper.a

# the build target executable:

	$(CC) $(CFLAGS) -o keyboard_events main.c keyboard_events.o hot_keys_events.o symbolic_hot_keys.o $(OS_X_LIBS) $(COCOA_HELPER_LIBRARY) $(AUTOMATION_SCRIPTS_LIBRARY)

	$(CC) $(CFLAGS) -c keyboard_events.c

	$(CC) $(CFLAGS) -c hot_keys_events.c

	$(CC) $(CFLAGS) -c symbolic_hot_keys.c
